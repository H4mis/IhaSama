FROM Pizza p
LEFT JOIN Pizzantaytteet pt ON p.pizzaid = pt.pizzaid
LEFT JOIN Tayte t ON t.tayteid = pt.tayteid
GROUP BY p.pizzaid
order by pizzaid, null;
